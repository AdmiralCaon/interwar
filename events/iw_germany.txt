add_namespace = iw_germany

# Kapp Putsch
country_event = {
	id = iw_germany.1
	title = iw_germany.1.t
	desc = iw_germany.1.d
	
	fire_only_once = yes
	
	trigger = {
		tag = GER
		has_completed_focus = GER_factory_councils_act
	}
	
	option = {
		name = iw_germany.1.a
		
		custom_effect_tooltip = kapp_seizes_control_of_government
		set_country_flag = kapp_putch_happened
		hidden_effect = {
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			create_country_leader = {
				name = "Wolfgang Kapp"
				desc = "POLITICS_WOLFGANG_KAPP_DESC"
				picture = "Portrait_Germany_Wolfgang_Kapp.dds"
				expire = "1965.1.1"
				ideology = centrism
				traits = {
				}
			}
			remove_ideas = {
				GER_friedrich_ebert
			}	
		set_party_name = { ideology = neutrality long_name = GER_reichswehr name = GER_reichswehr }
		country_event = { id = iw_germany.2 days = 2 }
		news_event = { id = iw_news.1 hours = 12 }
		set_cosmetic_tag = GER_kapp_germany
		}
	}
}

# Rathenau Agreement - France Agreed
country_event = {
	id = iw_germany.10
	title = iw_germany.10.t
	desc = iw_germany.10.d
	
	is_triggered_only = yes
	
	option = {
		name = iw_germany.10.a
		
		if = {
			limit = {
				has_idea = GER_treaty_of_versailles_national_disarmament_law_concessions
				}
			swap_ideas = {
				add_idea = GER_treaty_of_versailles_national_disarmament_law_concessions_rathenau
				remove_idea = GER_treaty_of_versailles_national_disarmament_law_concessions
			}
			else = {
				swap_ideas = {
					add_idea = GER_treaty_of_versailles_national_disarmament_law_rathenau
					remove_idea = GER_treaty_of_versailles_national_disarmament_law
				}
			}
		}
	}
}
